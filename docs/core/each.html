function each ( obj, iterator, context ) {
    if ( obj === null ) return;
    if ( nativeForEach && obj.forEach === nativeForEach ) {
        obj.forEach( iterator, context );
    } else if ( obj.length === +obj.length ) {

            if ( i in obj && iterator.call( context, obj[ i ], i, obj ) === breaker ) return;
