1 (function (W) {
  2 
  3         /** @namespace W.event
  4          **/
  5   	W.event = {};
  6 
  7          /**
  8          *  Create new W.event.Dispatcher
  9          *  
 10          *  Based on <a href="http://www.nonobtrusive.com/2009/07/24/custom-events-in-javascript-by-making-your-own-dispatcher-class/">Custom Event Dispatcher</a> by Erik Karlsson. Extended to pass a details custom object
 11          *
 12          *  @todo   bind events by default
 13          *  @todo   check for non-blocking implemetation
 14          *  @todo   implement bubbling
 15          *
 16          *  @example
 17          *
 18          *  function SomeClass(){
 19          *      Dispatcher.call(this);
 20          *  }
 21          *  SomeClass.prototype = new Dispatcher();
 22          *
 23          *  SomeClass.prototype.sendSomeEvent=function(){
 24          *      this.dispatch("test");
 25          *  }
 26          *
 27          *  var foo = new SomeClass();
 28          *  foo.addEventlistener( "test", function(context, data){ alert("bah"); } )
 29          *  foo.sendSomeEvent();
 30          *
 31          *
 32          *  @class
 33          */
 34         W.event.Dispatcher = function () {
 35             this.events=[];
 36         }
 37 
 38         /** Add event listener. Callback will recieve (context, details) arguments */
 39         W.event.Dispatcher.prototype.addEventlistener=function(/** String */ event, /** Function */ callback){
 40             this.events[event] = this.events[event] || [];
 41             if ( this.events[event] ) {
 42 		this.events[event].push(callback);
 43             }
 44         }
 45 
 46         /** Remove event listener
 47          *
 48          **/
 49         W.event.Dispatcher.prototype.removeEventlistener=function(/** String */ event, /** Function */ callback){
 50             if ( this.events[event] ) {
 51 		var listeners = this.events[event];
 52 		for ( var i = listeners.length-1; i>=0; --i ){
 53                     if ( listeners[i] === callback ) {
 54 			listeners.splice( i, 1 );
 55 			return true;
 56                     }
 57 		}
 58             }
 59             return false;
 60         }
 61 
 62         /** Dispatch event. Delegates will be passes the dispatcher context and a details
 63          *
 64          * @param {String} event        Name of the Event to be dispatched
 65          * @param {Object} details      Custom object passed to event handlers
 66          *
 67          * @example
 68          * 
 69          *      obj.addEventListener("event", function(context, data) {} );
 70          * 
 71          **/
 72         W.event.Dispatcher.prototype.dispatch=function(event, details){
 73             if ( this.events[event] ) {
 74 		var listeners = this.events[event], len = listeners.length;
 75 		while ( len-- ) {
 76                     listeners[len](this, details);	//callback with self
 77 		}
 78             }
 79         }

 81 })(W);
