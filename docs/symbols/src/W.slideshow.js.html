248             .method( "next",

250                  * @name next

254                     if (this._isChangingSlide) {
255                         W.w(".next() fail: slide is in the process of changing");
256                         return this;
257                     }

259                     if (this._currentImageIndex + 1 === this._activeAlbum.slides.length) {
260                         if (this._settings.loops === true)

355                 .method( "transitionSlide",
356                 /**
357                  * @name transitionSlide
358                  * @private
359                  * @params nextSlide
360                  * @methodOf W.slideshow.Controller
361                  */
362                 function(/** W.slideshow.Slide */ nextSlide) {
363                     this._nextSlide = nextSlide;
364                     this._$view.append(this._nextSlide.$view);

367                 })
368                 .method('transitionDidFinish',
369                 /**
370                  * Called when transitions have finished
371                  * @name transitionDidFinish
372                  * @private
373                  * @methodOf W.slideshow.Controller
374                  */
375                 function () {
376                     if(this._frontSlide == undefined) {
377                         this._frontSlide = this._nextSlide;
378                         this._slideDidChange();
379                     } else {
380                         this._frontSlide.$view.fadeOut(this._settings.transitionOutTime, W.bind(this, function () {
381                                 this._frontSlide = this._nextSlide;
382                                 this._slideDidChange();
383                             })
384                         );
385                     }
386                 });

425         /**
426             Creates a new  W.slideshow.Slide.
427             @class
428 
429                 @param (obj)  params    override/pass default information
430          */
431         W.slideshow.Slide = function(params) {
432             if (params == undefined)  params = {};
433 
434             /** url of image */
435             this.src =                params['src'] || "";
436             /** url of description */
437             this.description =        params['description'] || "";
438             /** url of title of image */
439             this.title  =             params['title'] || "";
440             /** a custom object */
441             this.details  =             params['details'] || {};
442             /** the DOM element. if not provided an div.img element will be created using `src` & `title`  */
443             this.$view =              params['$view'] || undefined;
444             /** style object for slide. default is 'position' : 'absolute',
445                                                                 'top' : 0,
446                                                                 'left' : 0  */
447             this.style =              params['style'] || {
448                                                                 'position' : 'absolute',
449                                                                 'top' : 0,
450                                                                 'left' : 0
451                                                            };
452 
453             // must be last
454             if (this.$view ==  undefined) this.renderView();
455 
456             return this;
457         };

459         W.slideshow.Slide
460             .method("renderView",
461                 /**
462                  * Creates the view element of the slide. Ideally this will not be called when a `$view` is provied at slide creation.
463                  * Will try to create an img tag using the `src` and title. Needs to be wrapped in a div
464                  * @name renderView
465                  * @returns W.slideshow.Slide
466                  * @methodOf W.slideshow.Slide
467                  */
468                  function () {
469                      this.$view = $('<div><img src="' + this.src + '" title="' + this.title + '" /></div>');
470                      this.$view.css(this.style);
471                      return this;
472                  }

476          /**
477             Creates a new  W.slideshow.Settings
478             @class
479          */
480         W.slideshow.Settings = function(params) {
481             if (params == undefined)     params = {};
482 
483             /**
484              * Overflow CSS for container
485              * @type    String
486              * @default overflow: hidden
487             */
488             this.overflow =              params['overflow'] || "hidden";
489             /**
490              * Milliseconds
491              * @type    Number
492              * @default 100
493             */
494             this.transitionInTime =        params['transitionInTime'] || 100;
495             /**
496              * Milliseconds. Only visible if transitioning out slide is visible (i.e. overlapping).
497              * @type    Number
498              * @default 100
499             */
500             this.transitionOutTime =        params['transitionOutTime'] || 20;
501             /**
502              * Where the gallery controller cycles through images
503              * @type    Boolean
504              * @default true
505             */
506             this.loops =                    params['loops'] || true;
507             /**
508              * Where the gallery controller cycles through images
509              * @type    Number
510              * @default 3000
511             */
512             this.slideDuration =                    params['slideDuration'] || 3000;
513         }
514         W.slideshow.Settings
515             .method("mergeSettings",
516                 /**
517                  * Safely merge settings
518                  * @private
519                  * @name mergeSettings
520                  * @methodOf W.slideshow.Controller
521                  */
522                 function (/** W.slideshow.Settings */ newSettings, /** W.slideshow.Settings */ oldSettings) {
523 
524                     // break if no arguments
525                     if (newSettings == undefined) return;
526                     // if no oldSettings create a setting object with defaults to merge with
527                     if (oldSettings == undefined) {
528                         oldSettings = new W.slideshow.Settings();
529                     };
530 
531                     // merge
532                     for (var i in newSettings) {
533                         oldSettings[i] = newSettings[i];
534                     };
535 
536                     return oldSettings;
537                 }
