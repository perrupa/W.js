 24 if(Function.prototype.method == undefined) {

 33 (function (W) {
 34 
 35         /**
 36          * @namespace W.route
 37          **/
 38   	W.route = {};
 39 
 40          /**
 41          *  Establishes a W.route.Router. Maps functions and, optionally, parameters to route strings<br />
 42          *  <br />
 43          *  Based on <a href="https://github.com/mtrpcic/pathjs">pathjs</a> by Mike Trpcic. Stripped down to not use the location.hash matching<br />
 44          *  <br />
 45          *  todo:   create a regex version more like backbone<br />
 46          *  todo:   allow for non URL strings<br />
 47          *  <br />
 48          *  <b>Route strings must be in the format /something/somthing/</b>
 49          *
 50          *  @example
 51          *
 52          *      W.route.Router.map("/test/route/:whatever/").to(function(){
 53          *            alert("test/route + " + this.params['whatever']);
 54          *      });
 55          *
 56          *      W.route.Router.dispatch("/test/route/helloworld");
 57          *
 58          *  @class
 59          */
 60         W.route.Router = {
 61             /**
 62              *  Create a route with path string.
 63              *  @param   {String}   path    Path string. For parameters use :Parameter e.g. /hello/:ross/
 64              *  @methodOf W.route.Router
 65              *  @name map
 66              *  @returns {W.route.Router.core.route} Route object
 67              **/
 68             'map': function (path) {
 69                 if ( W.route.Router.routes.hasOwnProperty(path)) {
 70                     return  W.route.Router.routes[path];
 71                 } else {
 72                     return new  W.route.Router.core.route(path);
 73                 }
 74             },
 75             'root': function (path) {
 76                  W.route.Router.routes.root = path;
 77             },
 78             'rescue': function (fn) {
 79                  W.route.Router.routes.rescue = fn;
 80             },
 81             'match': function (path) {
 82                 var params = {};
 83                 var route = null;
 84                 var i = null;
 85                 
 86                 for (route in  W.route.Router.routes) {
 87                 
 88                     if (route !== null && route !== undefined) {
 89                         var compare = path;
 90                         if (route.indexOf(":") > -1) {
 91                             for (i = 0; i < route.split("/").length; i++) {
 92                                 if ((i < compare.split("/").length) && (route.split("/")[i].charAt(0) === ":")) {
 93                                     params[route.split('/')[i].replace(":", '')] = compare.split("/")[i];
 94                                     compare = compare.replace(compare.split("/")[i], route.split("/")[i]);
 95                                 }
 96                             }
 97                         }
 98                         
 99                         
100 
101                         if (route === compare) {
102                             W.route.Router.routes[route].params = params;
103                             return  W.route.Router.routes[route];
104                         }
105                     }
106                 }
107 
108                 return null;
109             },
110             /**
111              *  Broadcast a Path string to the Route which will call any mapped functions to the provided route.
112              *  @param   {String}   path    Path string. For parameters use :ParameterName
113              *  @methodOf W.route.Router
114              *  @name dispatch
115              *  @returns {W.route.Router.core.route} Route object
116              **/
117             'dispatch': function (path) {
118                     var match =  W.route.Router.match(path);
119                     if (match !== null) {
120                         match.run();
121                     } else {
122                         if ( W.route.Router.routes.rescue !== null) {
123                             if ( W.route.Router.routes[ W.route.Router.routes.previous].do_exit !== null) {
124                                  W.route.Router.routes[ W.route.Router.routes.previous].do_exit();
125                             }
126 
127                              W.route.Router.routes.rescue();
128                         }
129                     }
130                 //}
131             },
132             'core': {
133                 'route': function (path) {
134                     this.path = path;
135                     this.action = null;
136                     this.do_enter = null;
137                     this.do_exit = null;
138                     this.params = {};
139                      W.route.Router.routes[path] = this;
140                 }
141             },
142             'routes': {
143                 'current': null,
144                 'root': null,
145                 'rescue': null,
146                 'previous': null
147             }
148         };
149 
150         /**
151          *  @name W.route.Router.core.route
152          *  @class
153          *  */
154          W.route.Router.core.route.prototype = {
155              /**
156              *  Bind a function to a route object
157              *  @param   {Function} fn  Function to be binded. The context containts a params collection of parameters found from map
158              *  @methodOf  W.route.Router.core.route
159              *  @name to
160              **/
161             'to': function (fn) {
162                 this.action = fn;
163                 return this;
164             },
165             'enter': function (fn) {
166                 this.do_enter = fn;
167                 return this;
168             },
169             'exit': function (fn) {
170                 this.do_exit = fn;
171                 return this;
172             },
173             'run': function () {
174                 if ( W.route.Router.routes.hasOwnProperty( W.route.Router.routes.previous)) {
175                     if ( W.route.Router.routes[ W.route.Router.routes.previous].do_exit !== null) {
176                          W.route.Router.routes[ W.route.Router.routes.previous].do_exit();
177                     }
178                 }
179 
180                 if ( W.route.Router.routes[this.path].hasOwnProperty("do_enter")) {
181                     if ( W.route.Router.routes[this.path].do_enter !== null) {
182                          W.route.Router.routes[this.path].do_enter();
183                     }
184                 }
185 
186                  W.route.Router.routes[this.path].action();
187             }
188         }
