 62 	/**
 63          * Asynchronous Loop
 64          *
 65          * @param   {Number}    iterations      Number of times to run
 66          * @param   {Function}  func            Function to execute in  iteration.
 67          *                                      Must call loop.next() when finished.
 68          * @param   {Function}  callback        On loop finshed call back
 69          *
 70    	 * @example
 71    	 * 		W.aloop(10, 
 72          *                      function (index, loop) {
 73          *                          log(index);
 74          *                          loop.next();
 75          *                      },
 76          *                      function () {
 77          *                          log('finished');
 78          *                      }
 79         *               );
 80    	*/
 81 	W.aloop = function (iterations, func, callback) {
 82 	    var index = 0;
 83 	    var done = false;
 84 	    var loop = {
 85 	        next: function() {
 86 	            if (done) {
 87 	                return;
 88 	            }
 89 
 90 	            if (index < iterations) {
 91 	                index++;
 92 	                func(index-1, loop);
 93 
 94 	            } else {
 95 	                done = true;
 96 	                callback();
 97 	            }
 98 	        },
 99 	        'break': function() {
100 	            done = true;
101 	            callback();
102 	        }
103 	    };
104 	    loop.next();
105 	    return loop;
106 	};

129 	W.text.addCommas = function (number) {
130 
131 		number = '' + number;
132 		if (number.length > 3) {
133 			var mod = number.length % 3;
134 			var output = (mod > 0 ? (number.substring(0,mod)) : '');
135 			for (i=0 ; i < Math.floor(number.length / 3); i++) {
136 				if ((mod == 0) && (i == 0))
137 					output += number.substring(mod+ 3 * i, mod + 3 * i + 3);
138 				else
139 					output+= ',' + number.substring(mod + 3 * i, mod + 3 * i + 3);
140 			}
141 			return (output);
142 		}
143 		else return number;

170     /**
171      *  Safe console. Aliased using W.l(); or W.log();
172      *  @param      {Object ...}    obj     Objects to log to console
173      **/
174     W.console.log = function () {
175         if(arguments.length > 0) {
176 
177             // join for graceful degregation
178             var args = (arguments.length > 1) ? Array.prototype.join.call(arguments, " ") : arguments[0];
179 
180             // this is the standard; firebug and newer webkit browsers support this
181             try {
182                     console.log(args);
183                     return true;
184             } catch(e) {
185                     // newer opera browsers support posting erros to their consoles
186                     try {
187                             opera.postError(args);
188                             return true;
189                     } catch(e) { }
190             }
191 
192             // catch all; a good old alert box
193             //alert(args);
194             return false;
195         }
