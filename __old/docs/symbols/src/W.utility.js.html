 69         // ## XML parser
 70 
 71         // Parse an XML-string.
 72         // Actually this is not a full implementation, but just
 73         // the basic parts to get it up running.
 74         // Nonetheless it is Good Enough(tm) for most uses.
 75         //
 76         // Known deficiencies: CDATA is not supported, will accept even
 77         // non-well-formed documents, <?... > <!... > are not really handled, ...
 78         exports.fromXml = function(xml) {
 79             if(typeof(xml) !== "string") {
 80                 JsonML_Error(
 81                     "Error: jsonml.parseXML didn't receive a string as parameter");
 82             }
 83 
 84             // white space definition
 85             var whitespace = " \n\r\t";
 86             // the current char in the string that is being parsed
 87             var c = xml[0];
 88             // the position in the string
 89             var pos = 0;
 90             // stack for handling nested tags
 91             var stack = [];
 92             // current tag being parsed
 93             var tag = [];
 94             // read the next char from the string
 95             function next_char() { c = ++pos < xml.length ? xml[pos] : undefined; }
 96             // check if the current char is one of those in the string parameter
 97             function is_a(str) { return str.indexOf(c) !== -1; }
 98             // return the string from the current position to right before the first
 99             // occurence of any of symb. Translate escaped xml entities to their value
100             // on the fly.
101             function read_until(symb) {
102                 var buffer = [];
103                 while(c && !is_a(symb)) {
104                     if(c === '&') {
105                         next_char();
106                         var entity = read_until(';');
107                         if(entity[0] === '#') {
108                             if(entity[1] === 'x') {
109                                 c = String.fromCharCode(parseInt(entity.slice(2), 16));
110                             } else {
111                                 c = String.fromCharCode(parseInt(entity.slice(1), 10));
112                             }
113                         } else {
114                             c = entities[entity];
115                             if(!c) {
116                                 JsonML_Error("error: unrecognisable xml entity: "
117                                              + entity);
118                             }
119                         }
120                     }
121                     buffer.push(c);
122                     next_char();
123                 }
124                 return buffer.join("");
125             }
126 
127             // The actual parsing
128             while(is_a(whitespace)) { next_char(); }
129             while(c) {
130                 if(is_a("<")) {
131                     next_char();
132 
133                     // `<?xml ... >`, `<!-- -->` or similar - skip these
134                     if(is_a("?!")) {
135                         if(xml.slice(pos, pos+3) === "!--") {
136                             pos += 3;
137                             while(xml.slice(pos, pos+2) !== "--") {
138                                 ++pos;
139                             };
140                         }
141                         read_until('>');
142                         next_char();
143 
144                     // `<sometag ...>` - handle begin tag
145                     } else if(!is_a("/")) {
146                         // read tag name
147                         var newtag = [read_until(whitespace+">/")];
148 
149                         // read attributes
150                         var attributes = {}
151                         var has_attributes = 0;
152                         while(c && is_a(whitespace)) { next_char(); };
153                         while(c && !is_a(">/")) {
154                             has_attributes = 1;
155                             var attr = read_until(whitespace + "=>");
156                             if(c === "=") {
157                                 next_char();
158                                 var value_terminator = whitespace+">/";
159                                 if(is_a('"\'')) { value_terminator = c; next_char(); }
160                                 attributes[attr] = read_until(value_terminator);
161                                 if(is_a('"\'')) next_char();
162                             } else {
163                                 JsonML_Error("something not attribute in tag");
164                             }
165                             while(c && is_a(whitespace)) { next_char(); };
166                         }
167                         if(has_attributes) { newtag.push(attributes); }
168 
169                         // end of tag, is it `<.../>` or  `<...>`
170                         if(is_a("/")) {
171                             next_char();
172                             if(!is_a(">")) {
173                                 JsonML_Error('expected ">" after "/" within tag');
174                             }
175                             tag.push(newtag);
176                         } else {
177                             stack.push(tag);
178                             tag = newtag;
179                         }
180                         next_char();
181 
182                     // `</something>` - handle end tag
183                     } else {
184                         next_char();
185                         if(read_until(">") !== tag[0]) {
186                             JsonML_Error("end tag not matching: " + tag[0]);
187                         }
188                         next_char();
189                         var parent_tag = stack.pop();
190                         if(tag.length <= 2 && !isArray(tag[1])
191                                            && typeof(tag[1]) !== "string") {
192                             tag.push("");
193                         }
194                         parent_tag.push(tag);
195                         tag = parent_tag;
196 
197                     }
198 
199                 // actual content / data between tags
200                 } else {
201                     tag.push(read_until("<"));
202                 }
203             }
204             return tag;
205         }

211         // Convert jsonml in array form to xml.
212         exports.toXml = function(jsonml) {
213             var acc = [];
214             toXmlAcc(jsonml, acc);
215             return acc.join('');
216         }

218         // The actual implementation. As the XML-string is built by appending to the
219         // `acc`umulator.
220         function toXmlAcc(jsonml, acc) {
221             if(isArray(jsonml)) {
222                 acc.push("<");
223                 acc.push(jsonml[0]);
224                 var pos = 1;
225                 var attributes = jsonml[1];
226                 if(attributes && !isArray(attributes)
227                               && typeof(attributes) !== "string") {
228                     for(var key in attributes) {
229                         acc.push(' ');
230                         acc.push(key);
231                         acc.push('="');
232                         xmlEscape(attributes[key], acc);
233                         acc.push('"');
234                     }
235                     ++pos;
236                 }
237                 if(pos < jsonml.length) {
238                     acc.push(">");
239                     {do {
240                         toXmlAcc(jsonml[pos], acc);
241                         ++pos;
242                     } while(pos < jsonml.length); }
243                     acc.push("</");
244                     acc.push(jsonml[0]);
245                     acc.push(">");
246                 } else {
247                     acc.push(" />");
248                 }
249             } else {
250                 xmlEscape(jsonml.toString(), acc);
251             }
252         }

255         var entities = {

257             "amp": '&',
258             "apos": "'",

264         var reventities = (function () {

266             for(var key in entities) {
267                 result[entities[key]] = key;

273         // Append the characters of `str`, or the xml-entity they map to, to the `acc`umulator array.
274         function xmlEscape(str, acc) {
275             for(var i = 0; i < str.length; ++i) {
276                 var c = str[i];
277                 var code = c.charCodeAt(0);
278                 var s = reventities[c];
279                 if(s) {
280                     acc.push("&" + s + ";");
281                 } else if(/*code < 32 ||*/ code >= 128) {
282                     acc.push("&#" + code + ";");
283                 } else {
284                     acc.push(c);
285                 }
286             }

314         // Convert jsonml into an easier subscriptable json structure, not preserving
315         // the order of the elements
316         exports.toObject = function(jsonml) {
317             var result = {};
318             result[jsonml[0]] = toObjectInner(jsonml);
319             return result;
320         }
321 
322         // Internal function called by toObject. Return an object corresponding to
323         // the child nodes of the `jsonml`-parameter
324         function toObjectInner(jsonml) {
325             var result = {};
326             var attr = jsonml[1];
327             var pos;
328             if(typeof(attr) === "object" && !isArray(attr)) {
329                 for(var key in attr) {
330                     result["@" + key] = attr[key];
331                 }
332                 pos = 2;
333             } else {
334                 pos = 1;
335                 if(jsonml.length === 2 && !isArray(attr)) {
336                     return attr;
337                 }
338             }
339             while(pos < jsonml.length) {
340                 var current = jsonml[pos];
341                 if(isArray(current)) {
342                     addprop(result, current[0], toObjectInner(current));
343                 } else {
344                     addprop(result, "_", current);
345                 }
346                 ++pos;
347             }
348             return result;
349         };

351         // Add a property to the object. If the property is already there, append
352         // the `val`ue to an array at the key instead, possibly putting existing
353         // object in front of such array, if that is not an array yet.
354         function addprop(obj, key, val) {
355             if(obj[key]) {
356                 if(isArray(obj[key])) {
357                     obj[key].push(val);
358                 } else {
359                     obj[key] = [obj[key], val];
360                 }
361             } else {
362                 obj[key] = val;
363             }
364         }
